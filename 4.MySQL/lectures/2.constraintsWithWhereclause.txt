	id int NOT NULL UNIQUE,
	Name varchar(50) NOT NULL,
	age INT NOT NULL CHECK(age>=18),
	gender VARCHAR(10) NOT NULL,
	phone VARCHAR(10) NOT NULL UNIQUE,
	city VARCHAR(10) NOT null DEFAULT 'Karachi'
);
